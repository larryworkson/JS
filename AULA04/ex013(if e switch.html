Condições

Condições simples

if (x > 3){
        console.log('A variável X é maior que 3')
    }

Condições compostas

        if (x == 'sim') {
            console.log('X é igual a 'sim')
    }
        else {
            console.log('x não é igual a 'sim')
        }

Condições aninhadas

        if (x === 20) {
            console.log('x tem igualdade restrita a 20')
        else if (x >= 20 && x != 30) {
            console.log('x é maior ou igual a 20 e não é 30')
        }
        else if (x < 20 || x == 0) {
            console.log('x é menor que 20 e igual a 0')
        }
    }

Condições multiplas (expressão switch case)

        var agora = new Date() //puxa o dia do sistema
        var diaSemana = agora.getDay()
        switch (diaSemana) {
            case 0:
                console.log('Domingo')
                break
            case 1:
                console.log('Segunda')
                break
            case 2:
                console.log('Terça')
                break
            case 3:
                console.log('Quarta')
                break
            case 4:
                console.log('Quinta')
                break
            case 5:
                console.log('Sexta')
                break
            case 6: 
                console.log('Sábado')
                break
            default:
                console.log('ERRO, dia inválido.')
                break
        }
